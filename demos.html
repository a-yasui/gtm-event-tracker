JavaScript

$('.track-it.click').gaeventtracker({       	        
  'ga_type' : 'all', 
  'category' : 'Click Fallback Category',
  'action' : 'Click Fallback Action',
  'value' : 1234,          				
  'nonint' : false,         				
  'delay' : true,
  'delay_timeout' : 100,
  'debug' : {
	'status' : true,
	'type' : 'alert'
	}                
});
