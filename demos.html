<script>
$('.track-it.hover').gaeventtracker({         
  'ga_type'  : 'all',
  'trigger'  : 'click',    
  'category' : 'Click Fallback Category',   
  'action'   : 'Click Fallback Action',     
  'value'    : 1111,          							
  'nonint'   : false,    
  'delay'    : true,     							
  'debug'    : {

    'type'   : 'alert'
  }                      
}); 
</script>
